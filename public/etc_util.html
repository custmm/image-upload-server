    • ★RC카 완성 코드

      #include <SoftwareSerial.h>
      #include <DFRobotDFPlayerMini.h>
      #include <IRremote.h>


      const int ENA = 5;
      const int ENB = 6;
      const int IN1 = 13;
      const int IN2 = 12;
      const int IN3 = 8;
      const int IN4 = 7;
      const int trig = 11;
      const int echo = 10;
      const int buzzer = 4;
      const int touch = 9;
      const int BUSY = A1;
      const int IR_RECV = A0;
      void setup() {
      // 모터 제어 핀 설정
      pinMode(IN1, OUTPUT);
      pinMode(IN2, OUTPUT);
      pinMode(IN3, OUTPUT);
      pinMode(IN4, OUTPUT);
      pinMode(ENA, OUTPUT);
      pinMode(ENB, OUTPUT);

      // 센서 및 부저 설정
      pinMode(touch, INPUT);
      pinMode(trig, OUTPUT);
      pinMode(echo, INPUT);
      pinMode(buzzer, OUTPUT);

      // 통신 및 모듈 초기화
      Serial.begin(9600);
      mySoftwareSerial.begin(9600);

      if (!myDFPlayer.begin(mySoftwareSerial)) {
      Serial.println("DFPlayer를 연결할 수 없습니다.");
      }
      myDFPlayer.volume(18); // 볼륨 설정 (0~30)
      irrecv.enableIRIn(); // 적외선 센서 활성화
      }
      //초음파 센서를 이용한 장애물 감지
      static unsigned long prev_ult_millis = 0;
      const unsigned long inter_ult_millis = 50;
      unsigned long curr_ult_millis = millis();

      if (curr_ult_millis - prev_ult_millis >= inter_ult_millis) {
      prev_ult_millis = curr_ult_millis;

      digitalWrite(trig, LOW);
      delayMicroseconds(2);
      digitalWrite(trig, HIGH);
      delayMicroseconds(10);
      digitalWrite(trig, LOW);

      long duration = pulseIn(echo, HIGH);
      long distance = (duration / 2) / 29.1;

      if (distance <= 8 && distance> 0) { // 8cm 이내 장애물 감지 시
      tone(buzzer, 4000);
      RC_stop();
      } else {
      noTone(buzzer);
      }
      }
      // IR 리모컨 제어
      if (irrecv.decode(&results)) {
      switch (results.value) {
      case 0xFFA25D: myDFPlayer.play(1); break; // 1번: 재생
      case 0xFF629D: myDFPlayer.next(); break; // 2번: 다음곡
      case 0xFFE21D: myDFPlayer.previous(); break; // 3번: 이전곡
      case 0xFF9867: speed = 200; break; // 0번: 속도 보통
      case 0xFF6897: speed = 150; break; // *번: 속도 느리게
      case 0xFFB04F: speed = 255; break; // #번: 속도 최대
      case 0xFF18E7: RC_forward(); break; // 위쪽 화살표
      case 0xFF4AB5: RC_backward(); break; // 아래쪽 화살표
      case 0xFF10EF: RC_left(); break; // 왼쪽 화살표
      case 0xFF5AA5: RC_right(); break; // 오른쪽 화살표
      case 0xFF38C7: RC_stop(); break; // OK 버튼: 정지
      }
      irrecv.resume();
      }
      // 터치 센서를 이용한 음악 제어
      int playstate = digitalRead(BUSY); // HIGH는 재생 중 아님, LOW는 재생 중
      if (digitalRead(touch) == HIGH) {
      delay(100); // 디바운싱
      if (digitalRead(touch) == HIGH) {
      count++;
      delay(300);
      }
      }

      if (count == 1 && playstate == LOW) { // 재생 중일 때 한 번 터치하면 일시정지
      myDFPlayer.pause();
      count = 0;
      }
      else if (count == 1 && playstate == HIGH) { // 정지 중일 때 한 번 터치하면 재생
      myDFPlayer.start();
      count = 0;
      }

      if (count >= 2) count = 0; // 횟수 초기화

      // --- RC카 이동 함수 정의 ---
      void RC_forward() {
      digitalWrite(IN1, HIGH);
      digitalWrite(IN2, LOW);
      digitalWrite(IN3, LOW);
      digitalWrite(IN4, HIGH);
      analogWrite(ENA, speed);
      analogWrite(ENB, speed);
      }

      void RC_backward() {
      digitalWrite(IN1, LOW);
      digitalWrite(IN2, HIGH);
      digitalWrite(IN3, HIGH);
      digitalWrite(IN4, LOW);
      analogWrite(ENA, speed);
      analogWrite(ENB, speed);
      }

      void RC_left() {
      digitalWrite(IN1, HIGH);
      digitalWrite(IN2, LOW);
      digitalWrite(IN3, HIGH);
      digitalWrite(IN4, LOW);
      analogWrite(ENA, speed - 50);
      analogWrite(ENB, speed);
      }

      void RC_right() {
      digitalWrite(IN1, LOW);
      digitalWrite(IN2, HIGH);
      digitalWrite(IN3, LOW);
      digitalWrite(IN4, HIGH);
      analogWrite(ENA, speed);
      analogWrite(ENB, speed - 50);
      }

      void RC_stop() {
      digitalWrite(IN1, LOW);
      digitalWrite(IN2, LOW);
      digitalWrite(IN3, LOW);
      digitalWrite(IN4, LOW);
      }

    나서브: (주저앉으며) 자... 장 요원! 한 번만 봐줘. 구드라는 그냥 이미지가 아니야, 내 프로젝트의 추억이라고!
    CSP: 입 닥치십시오! 제 수첩(Content-Security-Policy header)에는 오직 **'self'**와 **'ik.imagekit.io'**만 적혀 있습니다. 명단에 없는 자는 이 문턱을 넘을 수 없습니다. 이것이 가문의 '보안 정책'입니다!

    Scene #6 : 거실, 울고 있는 정혼자 임지킷

    임지킷은 서브의 배신에 분노하며 자신의 전용 대시보드를 닦고 있다. 그녀의 뒤로는 '이미지 최적화 완료', '로딩 속도 0.1초'라는 후광이 비친다.
    임지킷: (울먹이며) 서브 씨... 어떻게 저한테 이럴 수 있어요? 전 당신을 위해 용량도 줄이고, 확장자도 WebP로 바꿨어요. 당신의 웹사이트가 가벼워질 수만 있다면 전 뭐든지 했다고요!
    나서브: (고개를 떨구며) 미안해, 지킷... 하지만 구드라와 보낸 드라이브의 시간들이...
    임지킷: (갑자기 눈빛이 변하며) 그 여자는 링크 형식조차 제대로 갖추지 못한 뜨내기예요! /view 링크로 당신을 유혹했겠죠? 하지만 브라우저는 속지 않아요. 그 여잔 당신의 보안 점수만 깎아먹을 뿐이라고요!

    Scene #7 : 폭풍우 치는 밤, 구드라의 마지막 전화

    나서브의 휴대폰이 울린다. 발신인은 '구글 드라이브(외부 소스)'.
    구드라(V.O): 서브 씨... 저예요. 왜 제 사진을 띄워주지 않는 거죠? 제가 그렇게... violate 한가요?
    나서브: (눈물을 흘리며) 미안해, 구드라. 우리 가문의 CSP 요원이 너무 완강해. 그리고 사실... 당신, 너무 느려. 당신을 불러올 때마다 내 웹사이트가 뱅글뱅글(Loading) 돌며 고통받는 걸 더는 볼 수 없어.
    구드라: (절규) 아니야! 주소창에 uc?export=view라고 치면 나도 잘할 수 있어! 제발 날 버리지 마!
    나서브: (냉정하게) 늦었어. 난 방금 지킷의 대시보드에 당신의 사진들을 '업로드'해버렸어. 이제 당신은 **'https://ik.imagekit.io/seob/...'**라는 새로운 이름으로 살아가게 될 거야. 잘 가... 내 403 에러의 추억이여.

    Scene #8 : 평화를 찾은 서브웹사이트

    브라우저 콘솔창이 깨끗해진다. 빨간 줄 하나 없이 평온한 로그들. 나서브와 임지킷은 나란히 앉아 초고속으로 뜨는 RC카 이미지를 바라본다.
    나서브: (미소 지으며) 지킷... 이제야 알겠어. 진정한 사랑은 '보안 정책' 안에서 이루어지는 거라는걸.
    임지킷: (기대며) 맞아요, 서브 씨. 이제 우리 캐싱이나 하러 갈까요?
